## JavaScript Integration

### 1. Load Data
```javascript
let jobs = [];

async function loadJobs() {
  const response = await fetch('/data/jobs_final.json');
  jobs = await response.json();
  console.log(`Loaded ${jobs.length} jobs`);
}
```

### 2. Search Function (with aliases)
```javascript
function searchJobs(query) {
  const q = query.toLowerCase().trim();
  if (!q || q.length < 2) return [];
  
  const results = [];
  
  // Priority 1: Exact alias match
  for (const job of jobs) {
    const aliases = job.aliases || [];
    if (aliases.some(a => a.toLowerCase() === q)) {
      results.push({ ...job, matchType: 'alias_exact' });
    }
  }
  
  // Priority 2: Alias contains query
  for (const job of jobs) {
    if (results.some(r => r.id === job.id)) continue;
    const aliases = job.aliases || [];
    if (aliases.some(a => a.toLowerCase().includes(q))) {
      results.push({ ...job, matchType: 'alias_partial' });
    }
  }
  
  // Priority 3: Title contains query
  for (const job of jobs) {
    if (results.some(r => r.id === job.id)) continue;
    if (job.en.toLowerCase().includes(q) || job.de.toLowerCase().includes(q)) {
      results.push({ ...job, matchType: 'title' });
    }
  }
  
  return results.slice(0, 15);
}
```

### 3. Usage Examples
```javascript
// Modern job titles work via aliases:
searchJobs('ux designer')      // → User Interface Developer
searchJobs('frontend')         // → Mobile Developer
searchJobs('cto')              // → Technical Managing Director
searchJobs('partnership')      // → Business Development Manager
searchJobs('customer success') // → Account Manager
searchJobs('data engineer')    // → Data Scientist

// Original titles also work:
searchJobs('data scientist')   // → Data Scientist
searchJobs('software')         // → Multiple software jobs
```

### 4. Display Result
```jsx
function JobResult({ job }) {
  return (
    <div className="job-card">
      <h3>{job.en}</h3>
      <span className="sector">{job.sector}</span>
      
      <div className="profile-bars">
        <div className="bar human" style={{width: `${job.human}%`}}>
          Human-Centric {job.human}%
        </div>
        <div className="bar ai" style={{width: `${job.ai}%`}}>
          AI-Augmentable {job.ai}%
        </div>
        {job.auto > 0 && (
          <div className="bar auto" style={{width: `${job.auto}%`}}>
            Automation {job.auto}%
          </div>
        )}
      </div>
      
      <p className="dominant">
        Primarily: <strong>{job.dominant}</strong>
      </p>
    </div>
  );
}
```
